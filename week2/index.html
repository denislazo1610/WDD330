  • Array
    • const myArray = [] // To create array

myArray.pop(); // To remove last item of array
myArray.shift(); // To remove first item of array
myArray.push('value'); // Add a new value to end of array
myArray.unshift('value'); // Add a new value in start of array
  • Sets (no repeat value)
    • const list = new Set();
list.add(1); // To add a value to set
const list2 = new Set([1,2,3]); // Set with values in it
list.delete(1); // To delete a value 
list.clear(); // To remove all values 
const listArray = Array.from(list); // To convert set to array
  • Maps (with indexes)
    • const romanNumerals = new Map();
romanNumerals.set(1,'I'); // Add value 
romanNumerals.get(1) // I 
romanNumerals.remove(1) // Remove value
